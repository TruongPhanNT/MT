CREATE TABLE NIKAYA_CONTENT (
	ID_NIK INTEGER NOT NULL,
	ID_NIT INTEGER NOT NULL,
	ID_NIC INTEGER NOT NULL,
	NUMBER_NIC INTEGER,
	NAME_NIC_PAL VARCHAR(127),
	NAME_NIC_ENG VARCHAR(127),
	NAME_NIC_VIE VARCHAR(127),
	CONTENT_PAL BLOB SUB_TYPE BINARY,
	CONTENT_ENG BLOB SUB_TYPE BINARY,
	CONTENT_VIE BLOB SUB_TYPE BINARY,
	CONTENT_VIE_OTHER BLOB SUB_TYPE BINARY,
	CONTENT_PALI_VIET BLOB SUB_TYPE BINARY,
	CONSTRAINT PK_NIKAYA_CONTENT PRIMARY KEY (ID_NIK,ID_NIT,ID_NIC)
);
