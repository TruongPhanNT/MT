CREATE TABLE VINAYA_CONTENT (
	ID_VIT INTEGER NOT NULL,
	ID_VIC INTEGER NOT NULL,
	NAME_VIC_PAL VARCHAR(127),
	NAME_VIC_ENG VARCHAR(127),
	NAME_VIC_VIE VARCHAR(127),
	CONTENT_VIC_PAL BLOB SUB_TYPE BINARY,
	CONTENT_VIC_VIE BLOB SUB_TYPE BINARY,
	CONTENT_VIC_ENG BLOB SUB_TYPE BINARY,
	CONTENT_VIC_PAL_VIE BLOB SUB_TYPE BINARY,
	NUMBER_VIC INTEGER,
	CONSTRAINT PK_VINAYA_CONTENT PRIMARY KEY (ID_VIT,ID_VIC)
);

CREATE TABLE VINAYA_TREE (
	ID_VIT INTEGER NOT NULL,
	P_ID_VIT INTEGER,
	NAME_VIT_PAL VARCHAR(127),
	NAME_VIT_ENG VARCHAR(127),
	NAME_VIT_VIE VARCHAR(127),
	CONTENT_VIT BLOB SUB_TYPE BINARY,
	CONSTRAINT PK_VINAYA_TREE PRIMARY KEY (ID_VIT),
	CONSTRAINT FK_VINAYA_TREE_1 FOREIGN KEY (P_ID_VIT) REFERENCES VINAYA_TREE(ID_VIT) ON UPDATE CASCADE
);
